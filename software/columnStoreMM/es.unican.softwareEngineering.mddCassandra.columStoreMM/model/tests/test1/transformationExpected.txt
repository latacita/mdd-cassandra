
CREATE KEYSPACE Test1
WITH replication = {'class':'SimpleStrategy', 'replication_factor':1};

USE Test1;


CREATE TABLE users(
       idUsers text,
       given text,
       surname text,
       PRIMARY KEY(idUsers)
);

CREATE TABLE images(
       name text,
       owner text,
       date timestamp,
       tags set<text>,
       favs map<text,text>,
       PRIMARY KEY(name)
);

CREATE TABLE plays(
       idPlays text,
       game text,
       players int,
       scores list<int>,
       PRIMARY KEY(idPlays)
);
